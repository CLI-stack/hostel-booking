' + 'Your booking is now confirmed.</p>'; } else if (args && args.verifySuccess === false) { document.getElementById('verifyResultBody').innerHTML = '<i class="fas fa-times-circle" style="font-size:2.5rem;color:#dc2626"></i>' + '<h3 style="margin:12px 0 8px">Payment Failed</h3>' + '<p style="color:#475569">Verification failed. Please try again or contact support.</p>'; } PF('verifyResultDialog').show(); }
